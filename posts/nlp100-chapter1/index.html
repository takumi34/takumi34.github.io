def ngram(text, n):
    list = []
    tsize = len(text)
    for i in range(tsize - n + 1):
        list.append(text[i:i + n])
    return list

a = "I am an NLPer"
print(ngram(a, 2))  #文字bi-gram
print(ngram(a.split(' '), 2))  #単語bi-gram
  • 出力

def ngram(text, n):
    list = []
    tsize = len(text)
    for i in range(tsize - n + 1):
        list.append(text[i:i + n])
    return list


a = "paraparaparadise"
b = "paragraph"

abi = set(ngram(a, 2))
bbi = set(ngram(b, 2))

print(abi | bbi)  #和集合
print(abi & bbi)  #積集合
print(abi - bbi)  #差集合
print("ok" if "se" in (abi or bbi) else "no")

def cipher(s):
    res = []
    for i in s:
        if (i.islower()):
            res.append(chr(219 - ord(i)))
        else:
            res.append(i)
    return ''.join(res)


text = "chika chika"
a = cipher(text)
print(a)
b = cipher(a)
print(b)
  • 出力
xsrpz xsrpz
chika chika

import random


def typogly(s):
    if (len(s) <= 4):
        return s
    else:
        res = [s[0]] + random.sample(list(s[1:-1]), len(s[1:-1])) + [s[-1]]
        return "".join(res)
